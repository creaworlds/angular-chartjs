AngularJS ChartJS

Usage

  1. Include chartjs.js.
  2. Include angular-chartjs.js.
  3. Add directive 
    <html>
	<body ng-app="myApp">
		<div ng-controller="myController">
			<canvas ng-chart="myModel" width="200" height="200"></canvas>
		</div>
	</body>
</html>
  4. Inject some data and options 
    (function () {
	var myApp = angular.module("myApp", ["ngChart"]);
	myApp.controller("myController", ["$scope", function ($scope) {
		var data = {
			labels: ["January", "February", "March", "April", "May", "June", "July"],
			datasets: [
				{
					fillColor: "rgba(220,220,220,0.5)",
					strokeColor: "rgba(220,220,220,1)",
					pointColor: "rgba(220,220,220,1)",
					pointStrokeColor: "#fff",
					data: [65, 59, 90, 81, 56, 55, 40]
				},
				{
					fillColor: "rgba(151,187,205,0.5)",
					strokeColor: "rgba(151,187,205,1)",
					pointColor: "rgba(151,187,205,1)",
					pointStrokeColor: "#fff",
					data: [28, 48, 40, 19, 96, 27, 100]
				}
			]
		};
		$scope.myModel = { data: data, options: { animation: true } };
	}]);
})();
